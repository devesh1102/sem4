# 0+0 = 0
SDR 18 TDI(00000) 8 TDO(00) MASK(FF)
RUNTEST 1 MSEC
# 0+1 = 1
SDR 18 TDI(00001) 8 TDO(01) MASK(FF)
RUNTEST 1 MSEC
# 1+1 = 2
SDR 18 TDI(00101) 8 TDO(02) MASK(FF)
RUNTEST 1 MSEC
# 1+2 != 2
SDR 18 TDI(00102) 8 TDO(03) MASK(FF)
RUNTEST 1 MSEC
# 1<<3 = 8
SDR 18 TDI(30103) 8 TDO(08) MASK(FF)
RUNTEST 1 MSEC
# FF>>3 = 0F
SDR 18 TDI(2FF04) 8 TDO(0F) MASK(FF)
RUNTEST 1 MSEC
# -10--120 = 110
SDR 18 TDI(1F688) 8 TDO(6E) MASK(FF)
RUNTEST 1 MSEC
# 10-120 = -110
SDR 18 TDI(10A78) 8 TDO(92) MASK(FF)
RUNTEST 1 MSEC
# 10 << 3 = 80
SDR 18 TDI(30A03) 8 TDO(50) MASK(FF)
RUNTEST 1 MSEC
# 10 >>2 = 2
SDR 18 TDI(20A02) 8 TDO(02) MASK(FF)
RUNTEST 1 MSEC
# 10 +5 = 15
SDR 18 TDI(00A05) 8 TDO(0F) MASK(FF)
RUNTEST 1 MSEC
# 10 - 5 = 5
SDR 18 TDI(10A05) 8 TDO(05) MASK(FF)
RUNTEST 1 MSEC
